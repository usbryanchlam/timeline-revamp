```nginx
# ops/nginx/timeline.conf
#
# Timeline public reel — Nginx vhost with proxy_cache for the public surface.
#
# Phase 7 commits this file to the repo. Phase 8 symlinks it into
# /etc/nginx/conf.d/ on the OCI VM (D-19). TLS configuration is delegated to
# the Phase 8 certbot run — this file does not include `listen 443 ssl http2`
# or the certificate paths yet (they will be appended by certbot's --nginx
# installer plugin, or by a deploy-time include).
#
# Invalidation strategy: TTL-only, 5-minute windows (D-20).
# Cache key: $scheme$host$uri — no Vary header (D-21). English-only v1, locked.
#
# References:
#   - .planning/phases/07-public-urls-handle/07-CONTEXT.md §"Nginx cache" (D-18..D-21)
#   - .planning/phases/07-public-urls-handle/07-RESEARCH.md §"Pattern 7"
#   - server/routes/publicReel.ts — app-layer Cache-Control headers (300/60)
#   - server/routes/handlesCheck.ts — Cache-Control: no-store (NOT cached here)

# ---------------------------------------------------------------------------
# Cache zone declaration. Must live at the http{} level — Phase 8 places this
# file in conf.d/ which is included from http{} in nginx.conf, so this works.
# ---------------------------------------------------------------------------
proxy_cache_path /var/cache/nginx/public_reel
                 levels=1:2
                 keys_zone=public_reel:10m
                 max_size=1g
                 inactive=24h
                 use_temp_path=off;

# ---------------------------------------------------------------------------
# Upstream — Phase 8 may rewrite the address to the docker-compose service
# name (e.g. `api:3000`). Current placeholder is the local-VM port.
# ---------------------------------------------------------------------------
upstream timeline_api {
    server 127.0.0.1:3000;
}

server {
    # Phase 8 / certbot will add the 443 listen + SSL directives here.
    # Listen 80 stays for the certbot ACME http-01 challenge path.
    listen 80;
    server_name timeline.bryanlam.dev;

    # -----------------------------------------------------------------------
    # Public reel API — cached, no auth (07-02 mounts this OUTSIDE the JWT
    # middleware tree). Cache key is $scheme$host$uri — case-insensitive
    # handle lookup happens at the app layer, so /u/Bryan and /u/bryan are
    # different cache entries but return the same content. Acceptable
    # at-most-2x cache space per popular handle.
    # -----------------------------------------------------------------------
    location ~ ^/api/public/u/[^/]+$ {
        proxy_pass http://timeline_api;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_cache public_reel;
        proxy_cache_key $scheme$host$uri;
        proxy_cache_valid 200 5m;
        proxy_cache_valid 404 1m;

        # X-No-Cache request header bypasses cache for a single request.
        # Reserved for future owner-active invalidation (v2 — D-20 says
        # TTL-only for v1; this directive ships now so v2 doesn't need a
        # second Nginx edit).
        proxy_cache_bypass $http_x_no_cache;

        # Serve stale during upstream error / timeout / lock-driven update.
        # Combined with proxy_cache_lock, only one request per key
        # revalidates upstream at a time.
        proxy_cache_use_stale error timeout updating;
        proxy_cache_lock on;
        proxy_cache_lock_timeout 5s;

        # `always` is critical — without it, the header is omitted on 4xx/5xx
        # which defeats the debugging-on-404 use case.
        add_header X-Cache-Status $upstream_cache_status always;
    }

    # -----------------------------------------------------------------------
    # Public SPA HTML — same cache zone, same TTL. The HTML is identical
    # across handles (data comes from the API), so caching it here just
    # trims the upstream hop. try_files serves the static index.html via
    # the SPA fallback.
    #
    # Phase 8 may revisit if static-asset serving moves behind a separate
    # upstream (e.g. an S3-compat bucket) — for now, treat the SPA root
    # path identically to other locations.
    # -----------------------------------------------------------------------
    location ~ ^/u/[^/]+$ {
        try_files $uri /index.html;
    }

    # -----------------------------------------------------------------------
    # All other /api/* paths — pass through, NO proxy_cache.
    #
    # This catches:
    #   - /api/handles/check (Cache-Control: no-store from app — no caching
    #     needed; Nginx default also honors no-store)
    #   - /api/me, /api/cities, /api/photos (JWT-bearing — must never be
    #     cached; Authorization header forwarded)
    #   - /api/health (cheap; not worth caching)
    # -----------------------------------------------------------------------
    location /api/ {
        proxy_pass http://timeline_api;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_set_header Authorization $http_authorization;
        # No proxy_cache here — auth-bearing routes pass through.
    }

    # -----------------------------------------------------------------------
    # Static SPA fallback for all other paths.
    # -----------------------------------------------------------------------
    location / {
        try_files $uri /index.html;
    }
}
```

```bash
#!/usr/bin/env bash
# ops/nginx/timeline.conf.test.sh
#
# Lightweight self-check for ops/nginx/timeline.conf — verifies that every
# directive locked by Phase 7 CONTEXT.md (D-18 through D-21) is present.

#
# Optional: if `nginx` is on PATH, also runs `nginx -t -c ...` for full

#
# Exit code 0 = all checks pass. Exit code != 0 = at least one directive

set -euo pipefail

CONF="$(dirname "$0")/timeline.conf"

if [[ ! -f "$CONF" ]]; then
  echo "FAIL: $CONF does not exist"
  exit 1
fi

fail=0
check() {
  local desc="$1"
  local pattern="$2"
  if grep -qE "$pattern" "$CONF"; then
    echo "PASS: $desc"
  else
    echo "FAIL: $desc (pattern: $pattern)"
    fail=1
  fi
}

# D-18 directive set
check "proxy_cache_path declared" "^proxy_cache_path "
check "zone keys_zone=public_reel:10m" "keys_zone=public_reel:10m"
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-16 | Information Disclosure (negative-cache poisoning via crafted 404) | `location ~ ^/api/public/u/...` | mitigate | Cache key omits headers; 1-minute negative TTL bounds damage. App-layer `Cache-Control: public, max-age=60` matches the Nginx directive. |
| T-07-17 | Information Disclosure (sensitive headers cached) | Public reel location block | mitigate | Public route's DTO has no `Set-Cookie` or `Authorization` echo. The `/api/` catch-all block forwards Authorization but does NOT proxy_cache, so JWT-bearing responses are never persisted. |
| T-07-18 | Tampering (X-No-Cache request bypass abused for cache stampede) | `proxy_cache_bypass $http_x_no_cache` | accept | The bypass header skips cache for ONE request; it does NOT invalidate the existing entry. Worst case: an attacker forces N upstream hits. Acceptable at portfolio scale. Phase 8 may add rate-limit if abused. |
| T-07-19 | Tampering (cache poisoning via Vary mismatch) | Cache key | mitigate | No Vary header used; cache key is deterministic from URI. No way to poison a key with header-controlled variation. |
| T-07-20 | Information Disclosure (X-Cache-Status leaks cache state to attackers) | `add_header X-Cache-Status` | accept | Standard practice; useful for debugging at portfolio scale; the cache state (HIT/MISS/STALE) does not reveal any data secrets. |
| T-07-21 | Denial of Service (cache zone overflow) | `proxy_cache_path max_size=1g` | mitigate | 1GB zone with `inactive=24h` LRU eviction. At ~50KB per public reel payload, capacity is ~20k entries — far exceeds portfolio handle count. |
| T-07-22 | Spoofing (Nginx mis-symlinked or config drift between repo and live VM) | Phase 8 deploy | accept | Repo file is source of truth; Phase 8's symlink approach (D-19) prevents drift. `timeline.conf.test.sh` provides CI-verifiable integrity check. |
